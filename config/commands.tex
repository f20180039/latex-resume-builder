% Custom LaTeX Commands
% This file contains all custom command definitions used in the resume

% Strong text command with nested toggle capability
\makeatletter
\providecommand{\strong}[1]{}
\renewcommand{\strong}[1]{\@strong{#1}}
\newcommand{\@@strong}[1]{\textbf{\let\@strong\@@@strong#1}}
\newcommand{\@@@strong}[1]{\textnormal{\let\@strong\@@strong#1}}
\let\@strong\@@strong
\makeatother

% Section header with horizontal rule
\newcommand{\cvsection}[1]{
  \section*{\sc #1}
  \vspace{-1mm}
  \hrulefill
  \vspace{.12cm}
}

% Section header without horizontal rule
\newcommand{\specialcvsection}[1]{
  \section*{\sc #1}
}

% Environment definitions
\newenvironment{cvhonors}{\begin{itemize}}{\end{itemize}}
\newenvironment{cvitems}{\begin{itemize}}{\end{itemize}}
\newenvironment{cventries}{}{}

% CV honor entry (simplified)
\newcommand{\cvhonor}[4]{\item #1, #2}

% CV entry for experience/project sections
\newcommand{\cventry}[5]{
  {\small \strong{#2}} \hfill \emph{\footnotesize #3} \\
  {\footnotesize \emph{#1} \hfill #4}

  #5
  \vspace{1.2mm}
}

% Compact project entry with title and date on the same line
\newcommand{\projectentry}[4]{
  {\small \strong{#1}} \hfill {\footnotesize #2}%
  \if\relax\detokenize{#3}\relax
    \\[-1mm]
  \else
    \\[-0.5mm]{\footnotesize \emph{#3}}
  \fi

  #4
  \vspace{0.25mm}
}

% Small CV entry for education section
\newcommand{\smallcventry}[6]{
  {\small \strong{#2}} \hfill \emph{\footnotesize #5} \\
  {\footnotesize \emph{#1, #3} \hfill #4} \\
  \vspace{1.2mm}
}

% Custom column types for tables
\newcolumntype{L}[1]{>{\raggedright\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{C}[1]{>{\centering\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% Header generation macro using profile data
\newcommand{\makeheader}{
  \begin{minipage}[t]{\textwidth}
  {\Huge\textbf{\sc \href{\portfolio}{\fullname}}}
  \hfill
  {\footnotesize \href{mailto:\email}{\email\ \faEnvelope} \textbar{} \phone\ \faMobile}

  \jobtitle\ $|$ \education
  \hfill
  {\footnotesize \href{\linkedin}{\linkedinhandle\ \faLinkedin} \textbar{} \href{\portfolio}{Portfolio\ \faGlobe}}
  \end{minipage}
  \vspace{1mm}
}
